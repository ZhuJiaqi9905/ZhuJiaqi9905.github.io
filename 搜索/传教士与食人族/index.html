传教士与食人族

题目链接

传教士与食人族

题目描述

有M个传教士和野人。一个容量为N的船。他们想从A到B地，要保证无论是在A还是B还是船上，如果有传教士，那么那里的野人数一定不超过传教士数目。问最少要几步能完成任务。

解题思路

  • 此题可以用bfs，但是POJ有一个类似题直接bfs会RE，估计要A*算法。
  • 这个题难度在状态的表示，其实只需要记录A地的传教士数目，野人数目，和船的行驶方向即可。枚举可能在船上的人数情况，就能根据总人数直接推断出B地的人数。

代码

#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<vector>
#include<queue>
#include<string>
#include<cmath>
using namespace std;
struct Node {
    int m, c;//m是传教士数，c是野人数
    int step;
    int to;//0表示要从A到B，1表示要从B到A
};
bool visit[1010][1010][2];
int M;//M个cannibal和missionary
int N;//船的容量
queue<Node> q;
int main()
{
    scanf("%d%d", &M, &N);
    Node nd;
    nd.m = M; nd.c = M;
    nd.step = 0; nd.to = 0;
    q.push(nd); visit[M][M][0] = 1;
    int ans = -1;
    while (!q.empty()) {
        nd = q.front(); q.pop();
        if (nd.m == 0 && nd.c == 0 && nd.to == 1) {
            ans = nd.step;
            break;
        }
        int i, mcnt, ccnt;
        int boat;
        int k;
        int ma; int ca;
        int mb; int cb;
        Node tmp;
        if (nd.to == 0) {
            for (k = 1; k <= N; ++k) {//枚举船上的总人数
                for (mcnt = 0; mcnt <= k; ++mcnt) {//枚举船上的传教士数
                    ccnt = k - mcnt;
                    if (mcnt < ccnt && mcnt > 0) {
                        //船上有传教士且此时野人数大于传教士
                        continue;
                    }
                    ma = nd.m - mcnt;
                    ca = nd.c - ccnt;
                    mb = M - ma;
                    cb = M - ca;
                    if (ma < 0 || mb < 0 || ca < 0 || cb < 0) {
                        continue;
                    }
                    if (((ma > 0 && ma >= ca) || (ma == 0)) && ((mb > 0 && mb >= cb) || (mb == 0)) && !visit[ma][ca][1]) {
                        tmp.m = ma; tmp.c = ca; tmp.to = 1;
                        tmp.step = nd.step + 1; visit[ma][ca][1] = 1;
                        q.push(tmp);
                    }
                    
                }
            }
            
        }
        else {
            for (k = 1; k <= N; ++k) {
                for (mcnt = 0; mcnt <= k; ++mcnt) {//枚举船上的传教士数
                    ccnt = k - mcnt;
                    if (mcnt < ccnt && mcnt > 0) {
                        //船上有传教士且此时野人数大于传教士
                        continue;
                    }
                    ma = nd.m + mcnt;
                    ca = nd.c + ccnt;
                    mb = M - ma;
                    cb = M - ca;
                    if (ma < 0 || mb < 0 || ca < 0 || cb < 0) {
                        continue;
                    }
                    if (((ma > 0 && ma >= ca) || (ma == 0)) && ((mb > 0 && mb >= cb) || (mb == 0)) && !visit[ma][ca][0]) {
                        tmp.m = ma; tmp.c = ca; tmp.to = 0;
                        tmp.step = nd.step + 1; visit[ma][ca][0] = 1;
                        q.push(tmp);
                    }
                   
                }
            }
        }
    }
    printf("%d\n", ans);
    return 0;
}
