ZGOJ1561 区间抑或

题目链接

题目描述

解题思路

  • 利用线段树，区间操作注意用延迟更新，包括向下更新和向上更新。
  • 注意要用unsigned int。

#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<vector>
#include<queue>
#include<string>
#include<stack>
#include<cmath>
using namespace std;
int a[100010];
unsigned int b;
int n;//n个数 
int m;//m个查询 
struct Node {
	int L, R;
	int sum;//区间的抑或和 
	int lazy;
	int val;
	int Mid() {
		return (L + R) / 2;
	}
};
Node tree[400010];
void build_tree(int root, int l, int r)
{
	tree[root].L = l; tree[root].R = r; tree[root].lazy = 0; tree[root].val = 0;
	if (l == r) {
		tree[root].sum = a[l]; tree[root].val = a[l];
		return;
	}
	int lc = (root << 1);
	int rc = (root << 1) | 1;
	int mid = tree[root].Mid();
	build_tree(lc, l, mid);
	build_tree(rc, mid + 1, r);
	tree[root].sum = tree[lc].sum ^ tree[rc].sum;
}
int query_tree(int root, int l, int r)
{
	if (tree[root].L <= l && tree[root].R >= r && tree[root].lazy) {
		if ((r - l +1) % 2) {
			return tree[root].val;
		}
		else {
			return 0;
		}
	}
	if (tree[root].L == l && tree[root].R == r) {
		return tree[root].sum;
	}
	int lc = (root << 1);
	int rc = (root << 1) | 1;
	int mid = tree[root].Mid();
	if (r <= mid) {
		return query_tree(lc, l, r);
	}
	else if (l > mid) {
		return query_tree(rc, l, r);
	}
	else {
		return (query_tree(lc, l, mid) ^ query_tree(rc, mid + 1, r));
	}
}
void change_tree(int root, int l, int r, int num)
{
	if (tree[root].L == l && tree[root].R == r) {
		tree[root].val = num; tree[root].lazy = 1;
		if ((r - l +1 ) % 2) {
			tree[root].sum = num;
		}
		else
			tree[root].sum = 0;
		return;
	}
	int lc = (root << 1);
	int rc = (root << 1) | 1;
	int mid = tree[root].Mid();
	if (tree[root].lazy) {
		tree[root].lazy = 0; tree[lc].lazy = tree[rc].lazy = 1;
		tree[lc].val = tree[rc].val = tree[root].val;
		if ((tree[lc].R - tree[lc].L + 1) % 2) {
			tree[lc].sum = tree[lc].val;
		}
		else
			tree[lc].sum = 0;
		if ((tree[rc].R - tree[rc].L + 1) % 2) {
			tree[rc].sum = tree[rc].val;
		}
		else
			tree[rc].sum = 0;
	}
	if (r <= mid) {
		change_tree(lc, l, r, num);
	}
	else if (l > mid) {
		change_tree(rc, l, r, num);
	}
	else {
		change_tree(lc, l, mid, num);
		change_tree(rc, mid + 1, r, num);
	}
	tree[root].sum = tree[lc].sum ^ tree[rc].sum; //tree[root].lazy = 0;
	return;
}
int main()
{
	int T;
	scanf("%d", &T);
	int i, j; unsigned int ans;//注意要用unsigned int !!! 
	while (T--) {
		scanf("%d%d", &n, &m);
		memset(tree, 0, sizeof(tree));
		memset(a, 0, sizeof(a));

		for (i = 1; i <= n; ++i) {
			scanf("%u", &b);
			a[i] = (int)b;
		}
		
		build_tree(1, 1, n);
		char s[3]; int a, b, x;
		for (i = 1; i <= m; ++i) {
			scanf("%s", s);
			if (s[0] == 'C') {
				scanf("%d%d%d", &a, &b, &x);
				change_tree(1, a, b, x);
			}
			else {
				scanf("%d%d", &a, &b);
				ans = query_tree(1, a, b);
				printf("%u\n", ans);
			}
		}
		printf("\n");
	}
	return 0;
}
