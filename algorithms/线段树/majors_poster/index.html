POJ2528 Mayor's poster

题目链接

POJ2528 Mayor’s poster

题目描述

  • 用多张海报贴一面墙，求贴完后有几张海报是可见的（完全可见+部分可见）。
  • 题目有多组数据。
  • 对于每组数据有n个海报，每个海报给定左、右端点，并按照贴的顺序给出。

解题思路

  • 利用线段树，注意处理顺序是由外层到内层。因为外层覆盖的区间会影响内层的可见情况，而内层覆盖的区间不会影响外层的可见情况。
  • 注意延迟更新

代码

#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<vector>
#include<queue>
#include<string>
#include<stack>
#include<cmath>
using namespace std;
struct Node {
	int L, R;
	bool covered;
	int Mid()
	{
		return L + (R - L) / 2;
	}
};
struct Line {
	int s, e;
};

Line poster[10010];//记录海报
int disc[40010];//用于离散化
Node tree[160010];//线段树
int point[40010];//记录海报的两个端点，用于离散化
int point_num;//记录去重后的端点个数
void build_tree(int root, int l, int r);
bool query_tree(int root, int l, int r);
int main()
{
	int c; int n;//c个样例，n个海报
	scanf("%d", &c);
	int ans; int i, j;
	while (c--) {
		ans = 0;
		scanf("%d", &n);
		for (i = 0; i < n; ++i) {
			scanf("%d%d", &poster[i].s, &poster[i].e);
			point[2 * i] = poster[i].s;
			point[2 * i + 1] = poster[i].e;
		}
		sort(point, point + 2 * n);
		point_num = unique(point, point + 2 * n) - point;
		int dcnt = 0;
		disc[dcnt] = point[0]; dcnt++;
		//离散化，要把两个端点中间的区间也离散化成一个区间，
		//因此如果(point[i]-point[i-1]>1)，就把point[i-1]+1当一个新区间 
		for(i = 1; i < point_num; ++i){
			if( (point[i] - point[i-1]) > 1){
			disc[dcnt] = point[i-1]+1;	dcnt++; 
			}
			 disc[dcnt] = point[i]; dcnt++;
		}
		
		build_tree(1, 1, dcnt);
		int nl, nr;
		
		for(i = n - 1; i >= 0; --i){
			nl = lower_bound(disc, disc + dcnt, poster[i].s) - disc + 1;
			nr = lower_bound(disc, disc + dcnt, poster[i].e) - disc + 1;
			
			if(!query_tree(1, nl, nr) ){
				ans++;
			}
				
		} 
		printf("%d\n", ans);
	}
}


void build_tree(int root, int l, int r)
{
	tree[root].L = l; tree[root].R = r;
	tree[root].covered = 0;
	if (l == r)
		return;
	int mid = tree[root].Mid();
	build_tree((root << 1), l, mid);
	build_tree((root << 1) | 1, mid + 1, r);
	return;
}
bool query_tree(int root, int l, int r)
{
	bool flag = 0;
	if (tree[root].L == l && tree[root].R == r) {
		flag = tree[root].covered;
		tree[root].covered = 1;
		return flag;
	}
	if(tree[root].L <= l && tree[root].R >= r && tree[root].covered){
		//如果当前区间包裹了要查询的区间，且已经被覆盖了，就直接返回true,
		//因为你第一个是延迟更新，你这里不判断，就跑到下面了 
		//注意这里，错了两次了！！！ 
		return 1;
	}
	int mid = tree[root].Mid();
	int lc = (root << 1);
	int rc = (root << 1) | 1;
	if (r <= mid) {
		flag = query_tree(lc, l, r);
	}
	else if (l > mid) {
		flag = query_tree(rc, l, r);
	}
	else {
		flag = query_tree(lc, l, mid) & query_tree(rc, mid + 1, r);
	}
	//注意更新cover的状态
	tree[root].covered = tree[lc].covered & tree[rc].covered;
	return flag;
}
