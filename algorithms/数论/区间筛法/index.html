区间筛法

题目描述

求区间[a, b)内素数个数？其中a, b范围为1e12, b-a的范围是1e6

解题思路

  • 利用素数筛法。b以内的合数的最小质因数一定不超过sqrt(b)。因此在[2, sqrt(b))上筛素数的同时，也可将[a,b)中的合数筛去。
  • 因为题中a,b较大，所以构建[a, b)的表需要映射成0~b-a。
  • 题中的乘法需要long long范围，因此可将所有数都用long long范围。

代码

#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<vector>
#include<queue>
#include<string>
#include<stack>
#include<cmath>
using namespace std;

typedef long long ll;
const long long MAXN = 1e6 + 10;

bool is_prime[MAXN];//is_prime[i - a]判断i是否为素数，筛[a, b)
bool is_prime_small[MAXN];//is_prime_small[i]判断i是否为素数,筛[2, sqrt(b))

/*
对区间[a, b)内的整数进行筛法
*/
ll segment_sieve(ll a, ll b)
{
    for(ll i = 0; i * i < b; ++i){
        is_prime_small[i] = true;
    }
    for(ll i = 0, j = b - a; i < j; ++i){
        is_prime[i] = true;
    }
    for(ll i = 2; i * i < b; ++i){
        if(is_prime_small[i]){
            //筛[2, sqrt(b))
            for(ll j = 2*i; j * j < b; j += i){
                is_prime_small[j] = false;
            }
            //筛[a, b)
            for(ll j = max((a + i - 1)/i, 2LL)*i; j < b; j += i){
                is_prime[j - a] = false;
            }
        }
    }
    ll ans = 0;
    for(ll i = 0, j = b - a; i < j; ++i){
        if(is_prime[i])
            ans++;
    }
    return ans;
}
int main()
{
    ll a,  b;
    scanf("%lld%lld", &a, &b);
    ll ans = segment_sieve(a, b);
    printf("%lld\n", ans);
    return 0;
}
