银河英雄传说

题目链接

洛谷 银河英雄传说

题目描述

  • 见链接。实际上和Cube stacking类似。

解题思路

  • 类似于Cube stacking。利用并查集，维护sum[]数组和under[]数组。

代码

#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<vector>
#include<queue>
#include<string>
using namespace std;
const int MAX = 30000;
int parent[MAX + 10];//parent[i]是i的父节点
int sum[MAX + 10];//sum[i]是以i为根的集合的元素个数
int under[MAX + 10];//under[i]是i下面有的个数
int GetRoot(int a);
void Merge(int a, int b);
bool Query(int a, int b);
int main()
{
    for (int i = 1; i <= MAX; ++i) {
        parent[i] = i;
        sum[i] = 1;
        under[i] = 0;
    }
    int T = 0;
    scanf("%d", &T);


    char c[2]; int x; int y;
    while (T--) {

        scanf("%s%d%d", c, &x, &y);
        if (c[0] == 'M') {
            Merge(x, y);
        }
        else {
            if (Query(x, y)) {
                printf("%d\n", abs(under[x] - under[y]) - 1);
            }
            else {
                printf("-1\n");
            }
        }
    }
    return 0;
}
int GetRoot(int a)
{
    if (a == parent[a])
        return a;
    int ra = GetRoot(parent[a]);
    under[a] += under[parent[a]];
    parent[a] = ra;
    return ra;
}
void Merge(int a, int b)//把a放到b上
{
    int ra = GetRoot(a);
    int rb = GetRoot(b);
    parent[ra] = rb;
    under[ra] = sum[rb];
    sum[rb] += sum[ra];

    return;
}
bool Query(int a, int b)
{
    return GetRoot(a) == GetRoot(b);
}
